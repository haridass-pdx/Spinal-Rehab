from time

namelink , emp_id

SELECT CONCAT(employees.firstname,' ',employees.lastname) as fullname,
time.emp_id , sum(time.totaltime) as ttsubtotal,
sum(time.massagehrs) as mssubtotal
from time
INNER JOIN employees ON time.emp_id = employees.employee_id
WHERE time.time_date >= '2026/02/01' and time.time_date < '2026/02/16'
Group by
employees.firstname, employees.lastname , time.emp_id
order by
employees.firstname, employees.lastname , time.emp_id
